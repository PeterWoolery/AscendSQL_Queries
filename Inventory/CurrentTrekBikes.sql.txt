SELECT Locations.Name, Products.Description, Products.Quantity FROM Locations, Products, Categories, VendorProducts, Vendors WHERE Vendors.ID = VendorProducts.VendorID AND VendorProducts.ProductID = Products.ID AND Vendors.Name LIKE 'Trek%' AND Locations.ActiveStore = 1 AND Products.TopicID = Categories.ID AND Categories.Topic LIKE 'Bikes%' AND Products.Price > 1000 AND Products.Quantity > 0 AND Products.Hide <> 1 Order BY Products.Description
