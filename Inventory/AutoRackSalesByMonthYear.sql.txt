SELECT DATENAME(yy,Sales.FinishDate) as Yr, CASE DATEPART(m, Sales.FinishDate) WHEN 1 THEN 'January' WHEN 2 THEN 'February' WHEN 3 THEN 'March' WHEN 4 THEN 'April' WHEN 5 THEN 'May' WHEN 6 THEN 'June' WHEN 7 THEN 'July' WHEN 8 THEN 'August' WHEN 9 THEN 'September' WHEN 10 THEN 'October' WHEN 11 THEN 'November' WHEN 12 THEN 'December' END as Mth, SUM(SalesItems.Quantity) as [# of Sales], SUM(SalesItems.Price * SalesItems.Quantity) as Sales, SUM(SalesItems.Price * SalesItems.Quantity)/SUM(SalesItems.Quantity) as [Avg Sale] From Sales, Categories, Products, SalesItems WHERE Sales.ID = SalesItems.SalesID AND Products.ID = SalesItems.ProductID AND Categories.ID = Products.TopicID AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND Categories.Topic LIKE 'Accessories-Car Racks%' AND Sales.Hide = 0 AND SalesItems.Hide <> 1 AND Products.Hide <> 1 GROUP BY DATENAME(yy,Sales.FinishDate), DATEPART(m, Sales.FinishDate) ORDER BY  DATENAME(yy,Sales.FinishDate), DATEPART(m, Sales.FinishDate)
