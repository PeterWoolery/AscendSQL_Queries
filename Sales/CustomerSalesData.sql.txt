SELECT CONVERT(VARCHAR(10),cast(Sales.FinishDate as DATE),112) as [Sale Date],Customers.FirstName + ' ' + Customers.LastName + ' ' + Customers.Email as [Customer-Unique],Customers.Email AS Email,Customers.ZIP as [Zip Code],  CONVERT(DATE,Customers.DateCreated) AS [Customer Added] ,(DATEPART(dy,Sales.DateCreated)-DATEPART(dy,Customers.DateCreated)) AS CDATE,(DATEPART(dy,Sales.FinishDate)-DATEPART(dy,Sales.DateCreated)) AS Turnround, SalesItems.Quantity as Quantity ,(SalesItems.Quantity * SalesItems.Price) as Retail, ((SalesItems.Quantity * SalesItems.Price) - (SalesItems.Quantity * SalesItems.COGS)) AS Profit, (1-((SalesItems.Quantity * SalesItems.COGS)/(SalesItems.Quantity * SalesItems.Price))) as Margin,(Users.FirstName + ' ' + Users.LastName) AS [Sales Person], Categories.Topic, Sales.Barcode FROM SalesItems, Users,Products LEFT OUTER JOIN Categories ON Products.TopicID = Categories.ID,Sales LEFT OUTER JOIN Customers ON Sales.CustomerID = Customers.ID Where Users.ID <> 'efffffff-efff-effe-fffe-fffffffffffe' AND SalesItems.Hide = 0 AND Sales.Hide = 0 AND Sales.FinishDate BETWEEN CONVERT(DATE,GETDATE()) AND GETDATE()+1  AND Sales.ID = SalesItems.SalesID AND SalesItems.ProductID = Products.ID AND Sales.CommissionID = Users.ID AND SalesItems.Price <> 0 AND SalesItems.Quantity <> 0 ORDER BY Barcode
