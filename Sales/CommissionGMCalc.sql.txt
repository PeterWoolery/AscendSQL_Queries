SELECT Users.FirstName, SUM(SalesItems.Price*SalesItems.Quantity)-SUM(SalesItems.COGS*SalesItems.Quantity) AS [GM Commision], SUM(Sales.Total) FROM Sales, SalesItems, WorkOrderDetails, Users WHERE SalesItems.SalesID = Sales.ID AND Sales.Hide <> 1 AND Sales.Quote <> 1 AND SalesItems.Hide <> 1 AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND Users.ID = WorkOrderDetails.ServiceID AND Sales.ID = WorkOrderDetails.SalesID AND Sales.WorkOrder = 1 AND (SalesItems.WorkOrderDetailID = WorkOrderDetails.ID OR SalesItems.WorkOrderDetailID IS NULL) GROUP BY Users.FirstName
