SELECT Locations.Name, SUM(SalesItems.Quantity) AS [SOLD], SUM(OrderItems.Quantity) AS [Received], SUM(SalesItems.Quantity*SalesItems.Price) AS [Revenue], SUM(OrderItems.Quantity*OrderItems.Cost) AS [Total Cost] FROM Locations, SalesItems, Products, Sales, OrderItems, VendorProducts, Vendors, Orders Where Locations.ActiveStore = 1 AND SalesItems.SalesID = Sales.ID AND SalesItems.ProductID = Products.ID AND SalesItems.Hide <> 1 AND Sales.Hide <> 1 AND Products.ID = VendorProducts.ProductID AND Vendors.ID = VendorProducts.VendorID AND Vendors.Name LIKE 'Santa Cruz%' AND Sales.Hide <> 1 AND Sales.FinishDate BETWEEN GETDATE()-365 AND GETDATE() AND OrderItems.VendorProductID = VendorProducts.ID AND OrderItems.OrderID = Orders.ID AND OrderItems.Hide <> 1 AND Orders.Hide <> 1 AND OrderItems.DateReceived BETWEEN GETDATE()-365 AND GETDATE() AND Products.Price > 1000 GROUP BY Locations.Name
