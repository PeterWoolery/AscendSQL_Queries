SELECT Sales.ID, Payments.DateCreated as DateCreated ,Sales.BarCode AS [Transaction] ,Payments.PaymentType as PaymentType,Payments.CardType as CardType ,Payments.Amount as Amount ,Users.LastName + ', ' + Users.FirstName From Payments, Sales, Users Where Payments.SalesID = Sales.ID AND Sales.Hide = 0 AND Payments.Hide = 0 AND Payments.DateCreated BETWEEN DATEADD(day, -6, GETDATE()) AND DATEADD(day, +1, GETDATE()) AND Payments.CreatorID = Users.ID AND Payments.Amount < -100 ORDER BY Amount
