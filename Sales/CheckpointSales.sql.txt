SELECT Locations.Name, Products.Description, SUM(SalesItems.Quantity) From Locations, Products, SalesItems, Sales WHERE Sales.ID = SalesItems.SalesID AND Products.ID = SalesItems.ProductID AND Locations.ActiveStore = 1 AND Products.Description LIKE '%Checkpoint%' AND Sales.DateCreated BETWEEN GETDATE()-365 AND GETDATE() GROUP BY Locations.Name, Products.Description ORDER BY Products.Description
