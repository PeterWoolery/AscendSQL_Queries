SELECT Locations.Name, SUM(SalesItems.Quantity) AS [Bikes Sold], SUM(SalesItems.Quantity*SalesItems.Price) AS [Bike Dollars] FROM Locations, SalesItems, Sales, Products, Categories WHERE Locations.ActiveStore = 1 and SalesItems.ProductID = Products.ID AND SalesItems.SalesID = Sales.ID AND Products.TopicID = Categories.ID AND Sales.FinishDate BETWEEN '20190101' AND '20190731' AND Categories.Topic LIKE 'Bikes%' AND Sales.Quote <> 1 AND Sales.Hide <> 1 AND SalesItems.Hide <> 1 GROUP BY Locations.Name
