SELECT Users.FirstName, 

CONVERT(DECIMAL(6,1),(SELECT SUM(SalesItems.Quantity) FROM SalesItems, Sales, Products, Categories WHERE SalesItems.SalesID = Sales.ID AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND SalesItems.ProductID = Products.ID AND Categories.ID = Products.TopicID  AND Categories.Topic NOT LIKE 'Bikes%'  AND Categories.Topic NOT LIKE 'Labor%'  AND Categories.Topic NOT LIKE 'Rental%' AND Sales.CommissionID = Users.ID AND Sales.Barcode IN (SELECT Sales.Barcode FROM SalesItems, Sales, Products, Categories WHERE SalesItems.SalesID = Sales.ID AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND SalesItems.ProductID = Products.ID AND Categories.ID = Products.TopicID  AND Categories.Topic LIKE 'Bikes%' AND Sales.CommissionID = Users.ID)))/CONVERT(DECIMAL(6,1),(SELECT SUM(SalesItems.Quantity) FROM SalesItems, Sales, Products, Categories WHERE SalesItems.SalesID = Sales.ID AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND SalesItems.ProductID = Products.ID AND Categories.ID = Products.TopicID  AND Categories.Topic LIKE 'Bikes%' AND Sales.CommissionID = Users.ID AND SalesItems.Quantity > 0)) AS [Add-ons Per Bike], 

(SELECT SUM(SalesItems.Quantity) FROM SalesItems, Sales, Products, Categories WHERE SalesItems.SalesID = Sales.ID AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND SalesItems.ProductID = Products.ID AND Categories.ID = Products.TopicID  AND Categories.Topic LIKE 'Bikes%' AND Sales.CommissionID = Users.ID) AS [Bikes SOLD] 

FROM Users WHERE Users.ID IN (SELECT Users.ID FROM SalesItems, Sales, Products, Categories WHERE SalesItems.SalesID = Sales.ID AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND SalesItems.ProductID = Products.ID AND Categories.ID = Products.TopicID  AND Categories.Topic LIKE 'Bikes%' AND Sales.CommissionID = Users.ID AND SalesItems.Quantity > 0) ORDER BY Users.FirstName;
