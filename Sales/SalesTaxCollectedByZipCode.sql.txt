SELECT Customers.ZIP, Sum (SalesItems.Price*SalesItems.Quantity) as [Taxable Sales] From Customers, Sales, SalesItems, SalesItemTax where Sales.Hide = 0 AND Sales.FinishDate BETWEEN <<StartDate>> AND <<EndDate>> AND Sales.ID = SalesItems.SalesID AND SalesItems.Hide = 0 AND SalesItems.ID = SalesItemTax.SalesItemID AND Sales.CustomerID = Customers.ID AND SalesItemTax.Amount > 0  GROUP by Customers.ZIP Order by Customers.ZIP
