SELECT Sales.ID, CAST(Payments.DateCreated AS datetime) as DateCreated, CONVERT(char(10), Payments.DateCreated, 108) AS [Time], Sales.BarCode AS [Transaction] ,Payments.PaymentType as PaymentType,Payments.CardType as CardType ,Payments.Amount as Amount ,Users.LastName + ', ' + Users.FirstName From Payments, Sales, Users Where Payments.SalesID = Sales.ID AND Sales.Hide = 0 AND Payments.Hide = 0 AND Payments.DateCreated BETWEEN <<StartDate>> AND <<EndDate>> AND Payments.CreatorID = Users.ID ORDER BY Amount DESC
