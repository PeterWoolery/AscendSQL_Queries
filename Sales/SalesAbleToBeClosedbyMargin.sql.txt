Select Locations.Name, 

(SELECT SUM((SalesItems.Price-SalesItems.COGS)*SalesItems.Quantity) FROM Sales, SalesItems WHERE SalesItems.SalesID = Sales.ID AND Sales.FinishDate IS NULL AND Sales. QUOTE <> 1 AND Sales.Hide <> 1 AND Sales.Payments = Sales.Total AND SalesItems.Hide <> 1 AND Locations.ActiveStore = 1 AND Sales.ID NOT IN (SELECT Sales.ID FROM Sales, SalesItems, OrderItems WHERE Sales.ID = SalesItems.SalesID AND SalesItems.Backorder = 1 and Sales.Hide <> 1 AND Sales.Quote <> 1 AND SalesItems.Hide <> 1 AND OrderItems.ID = SalesItems.OrderItemID AND OrderItems.DateReceived IS NULL))  AS [Gross Margin Closeable],

(SELECT SUM(SalesItems.Price*SalesItems.Quantity) FROM Sales, SalesItems, WorkOrderDetails WHERE WorkOrderDetails.SalesID = Sales.ID AND Sales.WorkOrder = 1 AND WorkOrderDetails.DateCompleted IS NOT NULL AND SalesItems.Description LIKE 'LABOR%' AND SalesItems.SalesID = Sales.ID AND Sales.FinishDate IS NULL AND Sales. QUOTE <> 1 AND Sales.Hide <> 1 AND Sales.Payments = Sales.Total AND SalesItems.Hide <> 1 AND Locations.ActiveStore = 1 AND Sales.ID NOT IN (SELECT Sales.ID FROM Sales, SalesItems, OrderItems WHERE Sales.ID = SalesItems.SalesID AND SalesItems.Backorder = 1 and Sales.Hide <> 1 AND Sales.Quote <> 1 AND SalesItems.Hide <> 1 AND OrderItems.ID = SalesItems.OrderItemID AND OrderItems.DateReceived IS NULL))  AS [LABOR Closeable]

FROM Locations WHERE Locations.ActiveStore = 1;
