SELECT Locations.Name, Products.Description, SalesItems.Quantity, Products.MfgrPartNo, Categories.Topic FROM Locations, Sales, SalesItems, Products, Categories WHERE Locations.ActiveStore = 1 AND Sales.Hide <> 1 AND Sales.Quote <> 1 AND SalesItems.Hide <> 1 AND Sales.Payments > 50 AND Sales.ID = SalesItems.SalesID AND SalesItems.ProductID = Products.ID AND Products.TopicID = Categories.ID AND Categories.Topic LIKE 'Bike%' AND Sales.DateCreated BETWEEN '20190830' AND GETDATE() AND SalesItems.BackOrder <> 1 ORDER BY Categories.Topic, SalesItems.Description
